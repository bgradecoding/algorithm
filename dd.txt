    public static void main(String[] args){

        Scanner in = new Scanner(System.in);
        int str1 = in.nextInt();
        int str2 = in.nextInt();
        int answer = 0;

        Queue<Integer> queue = new Queue<>();
        for( int i = 0; i<str1; i++ ) queue.offer(i);

        while( queue.size() != 1 ){
            for(int i =0; i<str2; i++) queue.offer(queue.poll());
            queue.poll();
        }
        answer = queue.poll();

    }
